    <meta charset="UTF-8">

    <title>ELON Defender</title>

    <style>

        canvas { border: 1px solid black; background: #000; }

        body { display: flex; justify-content: center; align-items: center; height: 100vh; margin: 0; background: #111; color: white; font-family: Arial, sans-serif; }

        #game-over { display: none; position: absolute; text-align: center; background: rgba(0, 0, 0, 0.8); padding: 20px; border-radius: 10px; }

        #game-over button { padding: 10px 20px; font-size: 20px; cursor: pointer; background: #fff; border: none; border-radius: 5px; }

        #game-over button:hover { background: #ddd; }

        #win-screen { display: none; position: absolute; text-align: center; background: rgba(0, 255, 0, 0.8); padding: 20px; border-radius: 10px; }

        #win-screen button { padding: 10px 20px; font-size: 20px; cursor: pointer; background: #fff; border: none; border-radius: 5px; }

        #win-screen button:hover { background: #ddd; }

    </style>

<body>

    <canvas id="gameCanvas" width="800" height="600"></canvas>

    <div id="game-over">

        <h1>GAME OVER</h1>

        <h2>Final Score: <span id="final-score"></span></h2>

        <button onclick="restartGame()">Restart</button>

    </div>

    <div id="win-screen">

        <h1>YOU WIN!</h1>

        <h2>Final Score: <span id="win-score"></span></h2>

        <button onclick="restartGame()">Play Again</button>

    </div>

    <script>

        const canvas = document.getElementById('gameCanvas');

        const ctx = canvas.getContext('2d');

        const gameOverScreen = document.getElementById('game-over');

        const finalScoreEl = document.getElementById('final-score');

        const winScreen = document.getElementById('win-screen');

        const winScoreEl = document.getElementById('win-score');

        // Load assets

        const heroImg = new Image(); heroImg.src = 'Xlogo.jpg';

        const enemyImgs = [

            new Image(), new Image(), new Image()

        ];

        enemyImgs[0].src = 'EDMNLogo.png';

        enemyImgs[1].src = 'Walz.png';

        enemyImgs[2].src = 'NewMNFlag.png';

        const powerupImg = new Image(); powerupImg.src = 'heart.png';

        const shootSound = new Audio('laser-shoot.mp3');

        // Game objects

        let player = { x: 400, y: 300, size: 50, angle: 0, speed: 5, bullets: 1 };

        let bullets = [];

        let enemies = [];

        let powerups = [];

        let score = 0;

        let gameOver = false;

        let gameWon = false;

        // Controls

        let keys = {};

        window.addEventListener('keydown', e => keys[e.key] = true);

        window.addEventListener('keyup', e => keys[e.key] = false);

        // Spawn enemies

        function spawnEnemy() {

            if (gameOver || gameWon) return;

            const img = enemyImgs[Math.floor(Math.random() * enemyImgs.length)];

            const edge = Math.floor(Math.random() * 4);

            let x, y;

            if (edge === 0) { x = Math.random() * canvas.width; y = -50; } // Top

            else if (edge === 1) { x = canvas.width + 50; y = Math.random() * canvas.height; } // Right

            else if (edge === 2) { x = Math.random() * canvas.width; y = canvas.height + 50; } // Bottom

            else { x = -50; y = Math.random() * canvas.height; } // Left

            enemies.push({ x, y, size: 50, img, speed: 2 });

        }

        // Spawn powerups

        function spawnPowerup() {

            if (gameOver || gameWon) return;

            const x = Math.random() * (canvas.width - 30);

            const y = Math.random() * (canvas.height - 30);

            powerups.push({ x, y, size: 30 });

        }

        // Game loop

        function gameLoop() {

            if (gameOver || gameWon) {

                if (gameOver) {

                    gameOverScreen.style.display = 'block';

                    finalScoreEl.textContent = `$${score.toLocaleString()}`;

                } else {

                    winScreen.style.display = 'block';

                    winScoreEl.textContent = `$${score.toLocaleString()}`;

                }

                return;

            }

            // Clear canvas

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Player movement and rotation

            if (keys['ArrowLeft']) player.angle -= 0.1;

            if (keys['ArrowRight']) player.angle += 0.1;

            if (keys['ArrowUp']) {

                player.x += Math.cos(player.angle) * player.speed;

                player.y += Math.sin(player.angle) * player.speed;

            }

            if (keys['ArrowDown']) {

                player.x -= Math.cos(player.angle) * player.speed;

                player.y -= Math.sin(player.angle) * player.speed;

            }

            if (keys[' ']) shoot();

            // Keep player in bounds

            player.x = Math.max(player.size / 2, Math.min(canvas.width - player.size / 2, player.x));

            player.y = Math.max(player.size / 2, Math.min(canvas.height - player.size / 2, player.y));

            // Draw player

            ctx.save();

            ctx.translate(player.x, player.y);

            ctx.rotate(player.angle);

            ctx.drawImage(heroImg, -player.size / 2, -player.size / 2, player.size, player.size);

            ctx.restore();

            // Update and draw bullets

            bullets = bullets.filter(b => b.x > 0 && b.x < canvas.width && b.y > 0 && b.y < canvas.height);

            bullets.forEach(b => {

                b.x += b.vx;

                b.y += b.vy;

                ctx.fillStyle = 'white';

                ctx.beginPath();

                ctx.arc(b.x, b.y, 5, 0, Math.PI * 2);

                ctx.fill();

            });

            // Update and draw enemies

            enemies.forEach((e, ei) => {

                // Move towards player

                const dx = player.x - e.x;

                const dy = player.y - e.y;

                const dist = Math.sqrt(dx * dx + dy * dy);

                e.x += (dx / dist) * e.speed;

                e.y += (dy / dist) * e.speed;

                // Draw enemy

                ctx.drawImage(e.img, e.x - e.size / 2, e.y - e.size / 2, e.size, e.size);

                // Check collision with player

                if (dist < (player.size + e.size) / 2) {

                    gameOver = true;

                }

                // Check collision with bullets

                bullets.forEach((b, bi) => {

                    const bDist = Math.sqrt((b.x - e.x) ** 2 + (b.y - e.y) ** 2);

                    if (bDist < (e.size / 2 + 5)) {

                        enemies.splice(ei, 1);

                        bullets.splice(bi, 1);

                        score += 10000;

                        shootSound.play();

                        if (Math.random() < 0.2) spawnPowerup(); // 20% chance to spawn powerup

                    }

                });

            });

            // Update and draw powerups

            powerups.forEach((p, pi) => {

                ctx.drawImage(powerupImg, p.x - p.size / 2, p.y - p.size / 2, p.size, p.size);

                const dist = Math.sqrt((player.x - p.x) ** 2 + (player.y - p.y) ** 2);

                if (dist < (player.size + p.size) / 2) {

                    powerups.splice(pi, 1);

                    player.bullets += 1;

                }

            });

            // Draw score

            ctx.fillStyle = 'white';

            ctx.font = '20px Arial';

            ctx.fillText(`Score: $${score.toLocaleString()}`, 10, 30);

            ctx.fillText(`Bullets per shot: ${player.bullets}`, 10, 60);

            // Check win condition

            if (score >= 1000000000) {

                gameWon = true;

            }

            requestAnimationFrame(gameLoop);

        }

        // Shooting mechanics

        let lastShot = 0;

        function shoot() {

            const now = Date.now();

            if (now - lastShot < 300) return; // Shooting cooldown (300ms)

            lastShot = now;

            for (let i = 0; i < player.bullets; i++) {

                const angle = player.angle + (i - (player.bullets - 1) / 2) * 0.1; // Spread bullets

                bullets.push({

                    x: player.x,

                    y: player.y,

                    vx: Math.cos(angle) * 10,

                    vy: Math.sin(angle) * 10

                });

            }

        }

        // Restart game

        function restartGame() {

            gameOver = false;

            gameWon = false;

            gameOverScreen.style.display = 'none';

            winScreen.style.display = 'none';

            player = { x: 400, y: 300, size: 50, angle: 0, speed: 5, bullets: 1 };

            bullets = [];

            enemies = [];

            powerups = [];

            score = 0;

            gameLoop();

        }

        // Spawn enemies and powerups periodically

        setInterval(spawnEnemy, 2000);

        setInterval(spawnPowerup, 10000);

        // Start the game

        gameLoop();

    </script>
